Onnce we've got this working there is some stuff to do to start there service:

1 ---- system service file - /etc/systemd/system/flaskapp.service
[Unit]
Description=Flask App Service
After=network.target

[Service]
User=kryz-wosik
WorkingDirectory=/home/kryz-wosik/apps/Kryz-App-Main
ExecStart=/bin/bash -c 'source /home/kryz-wosik/apps/Kryz-App-Main/test_env/bin/activate && flask run --host=0.0.0.0'
Restart=always
EnvironmentFile=/home/kryz-wosik/apps/Kryz-App-Main/.flaskenv
Environment=APP_SETTINGS=/home/kryz-wosik/apps/Kryz-App-Main/settings.cfg

[Install]
WantedBy=multi-user.target

3 - these commands to start the service
sudo systemctl daemon-reload

sudo systemctl start flaskapp.service
sudo systemctl status flaskapp.service
